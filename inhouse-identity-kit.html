contrato visual

Copia esto antes de generar cualquier UI inhouse. Es el mínimo que define la marca.

inhouse-tokens.css
/* === INHOUSE · tokens canónicos (extraídos de los repos) === */
:root{
  /* marca raíz */
  --black:#000000; --copper:#d97736; --cream:#f5f5f0; --ink:#1a1a1a;
  /* producto (notes/learn) */
  --orange:#E07A3C;
  --n-yellow:#ffde00; --n-blue:#002FD9; --n-red:#E81010;
  --n-black:#4D4D4D; --n-green:#34c759; --green:#22c55e;
  /* proveedores IA */
  --ai-codex:#10a37f; --ai-gemini:#4285f4; --ai-claude:#c96442; --ai-gemma:#7c3aed;
  /* sub-marca math */
  --math-blue:#2563eb;
  /* superficie · radios · sombra */
  --surface:#ffffff; --border:rgba(0,0,0,.06);
  --r-md:14px; --r-card:2.5rem; --r-full:999px;
  --shadow-product:0 1px 3px rgba(0,0,0,.3), 0 4px 8px 3px rgba(0,0,0,.15);
  /* tipografía */
  --f-logo:'Comfortaa',cursive; --f-body:'DM Sans','Lato',sans-serif;
  --f-roboto:'Roboto',sans-serif; --f-mono:'Roboto Mono',monospace;
}
